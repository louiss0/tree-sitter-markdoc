=======================================
Blockquote with nested tag
=======================================

> {% callout %}
> Note
> {% /callout %}

---

(source_file
  (blockquote
    (blockquote_item
      (blockquote_marker)
      (blockquote_content
        (tag_open_block
          (tag_open)
          (tag_name)
          (tag_close
            (inline_expression_close)))))
    (blockquote_item
      (blockquote_marker)
      (blockquote_content
        (list_paragraph
          (text))))
    (blockquote_item
      (blockquote_marker)
      (blockquote_content
        (tag_end
          (tag_open)
          (tag_name)
          (inline_expression_close))))))
